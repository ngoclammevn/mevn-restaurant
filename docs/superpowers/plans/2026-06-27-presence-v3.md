# Presence v3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Three presence improvements — hide self from sidebar, grid centering + join ripple, cross-tab picks sync (merge union, realtime via Supabase Presence).

**Architecture:** All changes live in two files. `usePresence.js` gains `currentPicks` + `selfRemotePicks` state and `setMyPicks()` export. `MenuPage.vue` wires the new exports, filters self from viewer lists, and watches remote picks to merge locally. Supabase Presence is the transport for picks sync — no new channels.

**Tech Stack:** Vue 3 Composition API, Supabase Realtime Presence, CSS keyframes.

## Global Constraints

- No new npm packages
- No backend changes — RLS only, no service_role key in code
- Follow existing code style: no TypeScript annotations, `ref()` for reactive state
- Dish objects in `picks` come from MenuBoard's `@toggle-dish` event: `{ name: string, price: string, category: string, ... }`
- `menu.note` when structured is JSON: `{ dishes: Array<{ name, price, category, ... }>, notes?: string }`

```js
import { ref, onUnmounted, watch } from 'vue'
import { useUser } from '@clerk/vue'
import { useSupabaseClient } from '../lib/supabase'

const PERSON_COLORS = ['#e85d3a','#3b82f6','#f59e0b','#8b5cf6','#10b981','#ec4899','#f97316','#06b6d4']
const ANIMALS = ['Gấu','Voi','Hổ','Cáo','Thỏ','Cá heo','Rùa','Mèo','Bướm','Nai','Sói','Gà con']

export function getPersonColor(id) {
  if (!id) return PERSON_COLORS[0]
  let h = 0
  for (let i = 0; i < id.length; i++) h = (Math.imul(31, h) + id.charCodeAt(i)) | 0
  return PERSON_COLORS[Math.abs(h) % PERSON_COLORS.length]
}

function getAnonIdentity() {
  try {
    const s = sessionStorage.getItem('presence_anon')
    if (s) return JSON.parse(s)
  } catch {}
  const animal = ANIMALS[Math.floor(Math.random() * ANIMALS.length)]
  const id = crypto.randomUUID()
  const identity = { presenceKey: id, name: `${animal} ẩn danh`, color: getPersonColor(id), isAnon: true }
  try { sessionStorage.setItem('presence_anon', JSON.stringify(identity)) } catch {}
  return identity
}

export function usePresence(menuId) {
  const sb = useSupabaseClient()
  const { user, isLoaded } = useUser()
  const viewers = ref([])
  const isMorphingIn = ref(false)
  const myActiveDish = ref(null)
  const currentPicks = ref([])
  const selfRemotePicks = ref([])
  let channel = null
  let dishTimer = null
  let myKey = null

  function getMyPayload() {
    const anon = getAnonIdentity()
    if (user.value) {
      return {
        presenceKey: user.value.id,
        name: user.value.fullName || 'Bạn',
        avatar: user.value.imageUrl || null,
        color: getPersonColor(user.value.id),
        isAnon: false,
        activeDish: myActiveDish.value,
        picks: currentPicks.value,
      }
    }
    return { ...anon, activeDish: myActiveDish.value, picks: currentPicks.value }
  }

  async function connect() {
    if (channel) return
    const anon = getAnonIdentity()
    myKey = user.value?.id ?? anon.presenceKey
    channel = sb.channel(`menu-presence:${menuId}`, { config: { presence: { key: myKey } } })
    channel
      .on('presence', { event: 'sync' }, () => {
        const state = channel.presenceState()

        // Extract own remote picks (union across all slots for myKey, before dedup)
        const mySlots = state[myKey] || []
        selfRemotePicks.value = [...new Set(mySlots.flatMap(s => s.picks || []))]

        // Deduplicate viewers by presenceKey, merging activeDish and picks across slots
        const byKey = new Map()
        for (const entries of Object.values(state)) {
          for (const entry of entries) {
            const existing = byKey.get(entry.presenceKey)
            if (!existing) {
              byKey.set(entry.presenceKey, { ...entry, picks: [...(entry.picks || [])] })
            } else {
              if (entry.activeDish && !existing.activeDish) existing.activeDish = entry.activeDish
              if (entry.picks?.length) {
                const merged = new Set([...existing.picks, ...entry.picks])
                existing.picks = [...merged]
              }
            }
          }
        }
        viewers.value = [...byKey.values()]
      })
      .subscribe(async (status) => {
        if (status === 'SUBSCRIBED') await channel.track(getMyPayload())
      })
  }

  function setActiveDish(dishName) {
    myActiveDish.value = dishName || null
    if (dishTimer) clearTimeout(dishTimer)
    dishTimer = setTimeout(async () => {
      if (channel) await channel.track(getMyPayload())
    }, 300)
  }

  function setMyPicks(names) {
    currentPicks.value = names
    if (channel) channel.track(getMyPayload())
  }

  function handleVisibility() {
    if (!document.hidden && channel) channel.track(getMyPayload())
  }

  // Connect once Clerk loads
  watch(isLoaded, (loaded) => { if (loaded && !channel) connect() }, { immediate: true })

  // Re-connect when user signs in (anon → real user)
  watch(user, async (newUser, oldUser) => {
    if (!newUser || oldUser || !channel) return
    isMorphingIn.value = true
    selfRemotePicks.value = []
    setTimeout(() => { isMorphingIn.value = false }, 1600)
    await sb.removeChannel(channel)
    channel = null
    connect()
  })

  document.addEventListener('visibilitychange', handleVisibility)

  onUnmounted(() => {
    document.removeEventListener('visibilitychange', handleVisibility)
    if (dishTimer) clearTimeout(dishTimer)
    if (channel) sb.removeChannel(channel)
  })

  return { viewers, setActiveDish, isMorphingIn, getPersonColor, setMyPicks, selfRemotePicks }
}
```
